#Обязательный параметр. Имя правильной "структуры" Oracle = имя "структуры" пользователя+$id
#$id соответственно нужно использовать и в правильном запросе
$id=_temp
@teacher
CREATE TABLE bus1_temp (
       bus_number           VARCHAR2(20) NOT NULL,
       bus_id               INTEGER NOT NULL,
       cod_mark             INTEGER NOT NULL
);

#В этом параметре должен хранится код с помощью которого можно запустить или изменить
#состояние пользовательской и правильной "структур"
@execstr
insert table bus1_temp(bus_number,bus_id,cod_mark) value('номер',1,1);

#Запрос с помощью которого можно определить правильность "структуры" пользователя
